country_event = {
    id = flavor_cnnat.30040
    title = "flavor_cnnat.1t"
    desc = "flavor_cnnat.1d"
    picture = EXPLORERS_eventPicture
    hidden = yes
    
    trigger = {
        NOT = { has_country_flag = tc_african_great_lakes }
        3004 = {
            NOT = { has_province_flag = iscity }
        }
        is_colonial_nation = no
        calc_true_if = {
            any_core_province = {
                province_group = colonial_african_great_lakes_region
                owned_by = ROOT
                is_overseas = yes
            }
            amount = 1
        }
    }
    
    immediate = {
        3004 = { set_province_flag = iscity }
        save_global_event_target_as = cn_african_great_lakes
        every_subject_country = {
            limit = {
                event_target:cn_african_great_lakes = {
                    tag = ROOT
                }
                capital_scope = {
                    province_group = colonial_african_great_lakes_region
                }
            }
            clear_global_event_target = cn_african_great_lakes
            save_global_event_target_as = cn_african_great_lakes
        }
        
        if = {
            limit = {
                event_target:cn_african_great_lakes = {
                    tag = ROOT
                }
            }
            clear_global_event_target = cn_african_great_lakes
            country_event  = {
                id = flavor_cn.3004
            }
        }
        else = {
            3004 = { clr_province_flag = iscity }
            event_target:cn_african_great_lakes = {
                country_event  = {
                    id = flavor_cnnat.30042
                }
            }
            clear_global_event_target = cn_african_great_lakes
        }
    }
    
    option = {
        #hidden
    }
}

country_event = {
    id = flavor_cnnat.30350
    title = "flavor_cnnat.1t"
    desc = "flavor_cnnat.1d"
    picture = EXPLORERS_eventPicture
    hidden = yes
    
    trigger = {
        NOT = { has_country_flag = tc_comorin_cape }
        3035 = {
            NOT = { has_province_flag = iscity }
        }
        is_colonial_nation = no
        calc_true_if = {
            any_core_province = {
                province_group = colonial_comorin_cape_region
                owned_by = ROOT
                is_overseas = yes
            }
            amount = 1
        }
    }
    
    immediate = {
        3035 = { set_province_flag = iscity }
        save_global_event_target_as = cn_comorin_cape
        every_subject_country = {
            limit = {
                event_target:cn_comorin_cape = {
                    tag = ROOT
                }
                capital_scope = {
                    province_group = colonial_comorin_cape_region
                }
            }
            clear_global_event_target = cn_comorin_cape
            save_global_event_target_as = cn_comorin_cape
        }
        
        if = {
            limit = {
                event_target:cn_comorin_cape = {
                    tag = ROOT
                }
            }
            clear_global_event_target = cn_comorin_cape
            country_event  = {
                id = flavor_cn.3035
            }
        }
        else = {
            3035 = { clr_province_flag = iscity }
            event_target:cn_comorin_cape = {
                country_event  = {
                    id = flavor_cnnat.30352

            clear_global_event_target = cn_comorin_cape

    }
    
    option = {
        #hidden
    }
}
