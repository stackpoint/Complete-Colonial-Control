# Colonial Nation
country_event = {
	id = flavor_cn.3011
	title = "flavor_cn.3011"
	desc = "flavor_cn.DESC"
	picture = EXPLORERS_eventPicture
	
	trigger = {
		is_colonial_nation = no
	    NOT = { has_country_flag = colonized_mexico }
		3011 = {
		    is_city = no
		}
		calc_true_if = {
			any_core_province = {
				province_group = colonial_mexico_region
				owned_by = ROOT
				is_overseas = yes
			}
			amount = 1
		}
	}
	
	immediate = { set_country_flag = colonized_mexico }
	
	option = {			# Make a colonial nation
		name = "flavor_cn.option1"
		ai_chance = { 
 		factor = 70 
			modifier = { 
				factor = 0 
				has_global_flag = no_ai_colonial 
			} 	
 		}
		add_adm_power = 10
		add_prestige = 5
		custom_tooltip = cn_toopltip_option1
		hidden_effect = {
			3011 = {
				 cede_province = ROOT
				 add_core = ROOT
			}
		}
	}
	
	option = {			# All for us
		name = "flavor_cn.option2"
		ai_chance = { 
 		factor = 30 
			modifier = { 
				factor = 0 
				has_global_flag = all_ai_colonial 
			} 	
 		}
		add_adm_power = -50
		add_prestige = -10
		custom_tooltip = cn_toopltip_option2
	}
}
